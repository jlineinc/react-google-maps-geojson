# react-google-maps-geojson

>

## Install

```bash
npm install --save react-google-maps-geojson
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-google-maps-geojson'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
